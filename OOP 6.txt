CLASS MAGE INHERITS CHARACTER
 ATTRIBUTES
    	PRIVATE MANA: NUMBER

   CONSTRUCTOR(N:TEXT, H:NUMBER, INV:INVENTORY, M:NUMBER)
    	SUPER(N, H, INV)
    	SET MANA TO M
   END CONSTRUCTOR

 METHOD
    	OVERRIDE ATTACK()
        IF MANA >= 10 THEN
            SET MANA TO MANA - 10
            DISPLAY "Mage casts a magic spell."
        ELSE
            DISPLAY "Not enough mana."
        END IF
    	END METHOD
END CLASS

   CONSTRUCTOR(N:TEXT, H:NUMBER, INV:INVENTORY, A:NUMBER)
    	SUPER(N, H, INV)
    	SET ARROWS TO A
   END CONSTRUCTOR

 METHOD
    	OVERRIDE ATTACK()
        IF ARROWS > 0 THEN
          SET ARROWS TO ARROWS - 1
          DISPLAY "Archer shoots an arrow."
        ELSE
            DISPLAY "No arrows remaining."
        END IF
        END METHOD
END CLASS

	warrior1.ATTACK()
	mage1.ATTACK()
	archer1.ATTACK()

	DISPLAY "Character Inventory:" + inventory1.DISPLAY_ITEMS()

CLASS ELECTRICITY_BILL INHERITS BILL
    ATTRIBUTES
        PRIVATE RATE: NUMBER
        PRIVATE TIME_OF_DAY: TEXT

    CONSTRUCTOR(BP:TEXT, DD:TEXT, M:METER_READING, R:NUMBER, T:TEXT)
        SUPER(BP, DD, M)
        SET RATE TO R
        SET TIME_OF_DAY TO T
    END CONSTRUCTOR

    METHOD
        OVERRIDE CALCULATE_AMOUNT()

        END METHOD

CLASS METER_READING
    ATTRIBUTES
        PRIVATE START_READING: NUMBER
        PRIVATE END_READING: NUMBER

    CONSTRUCTOR(SR:NUMBER, ER:NUMBER)
        SET START_READING TO SR
        SET END_READING TO ER
    END CONSTRUCTOR

    METHOD
        GET_CONSUMPTION()
            RETURN END_READING - START_READING
        END METHOD
END CLASS

CLASS RIDE

        PRIVATE PICKUP_LOCATION: TEXT
        PRIVATE DROPOFF_LOCATION: TEXT
        PRIVATE ROUTE: ROUTE

    CONSTRUCTOR(P:TEXT, D:TEXT, R:ROUTE)
        SET PICKUP_LOCATION TO P
        SET DROPOFF_LOCATION TO D
        SET ROUTE TO R
    END CONSTRUCTOR

    METHOD
        CALCULATE_FARE()
            DISPLAY "Fare depends on ride type."
        END METHOD
END CLASS


CLASS STANDARD_RIDE INHERITS RIDE
    ATTRIBUTES
        PRIVATE DISTANCE: NUMBER
        PRIVATE RATE_PER_KM: NUMBER

    CONSTRUCTOR(P:TEXT, D:TEXT, R:ROUTE, DIS:NUMBER, RATE:NUMBER)
        SUPER(P, D, R)
        SET DISTANCE TO DIS
        SET RATE_PER_KM TO RATE
    END CONSTRUCTOR

    METHOD
        OVERRIDE CALCULATE_FARE()
            SET FARE TO DISTANCE * RATE_PER_KM
            RETURN FARE
        END METHOD
END CLASS


CLASS PREMIUM_RIDE INHERITS RIDE
  ATTRIBUTES
        PRIVATE DISTANCE: NUMBER
        PRIVATE HIGHER_RATE: NUMBER
        PRIVATE COMFORT_SURCHARGE: NUMBER

    CONSTRUCTOR(P:TEXT, D:TEXT, R:ROUTE, DIS:NUMBER, HR:NUMBER, CS:NUMBER)
        SUPER(P, D, R)
        SET DISTANCE TO DIS
        SET HIGHER_RATE TO HR
        SET COMFORT_SURCHARGE TO CS
    END CONSTRUCTOR

    METHOD
        OVERRIDE CALCULATE_FARE()
            SET FARE TO (DISTANCE * HIGHER_RATE) + COMFORT_SURCHARGE
            RETURN FARE
        END METHOD
END CLASS


CLASS SHARED_RIDE INHERITS STANDARD_RIDE
  ATTRIBUTES
        PRIVATE PASSENGERS: NUMBER

    CONSTRUCTOR(P:TEXT, D:TEXT, R:ROUTE, DIS:NUMBER, RATE:NUMBER, PASS:NUMBER)
        SUPER(P, D, R, DIS, RATE)
        SET PASSENGERS TO PASS
    END CONSTRUCTOR

    METHOD
        OVERRIDE CALCULATE_FARE()
            SET FARE TO (DISTANCE * RATE_PER_KM) / PASSENGERS
            RETURN FARE
        END METHOD
END CLASS


CLASS ROUTE
    ATTRIBUTES
        PRIVATE WAYPOINTS: LIST<TEXT>[]

    CONSTRUCTOR()
        SET WAYPOINTS TO LIST[]
    END CONSTRUCTOR

    METHOD
        ADD_WAYPOINT(W:TEXT)
            ADD W TO LIST WAYPOINTS
        END METHOD

        DISPLAY_WAYPOINTS()
            REPEAT FROM I = 1 TO SIZE OF WAYPOINTS
                DISPLAY WAYPOINTS[I]
            END REPEAT
        END METHOD
END CLASS

    	CREATE OBJECT SH1 OF SHARED_RIDE("Airport", "Hotel", R1, 20, 0.5, 4)

    	DISPLAY "Standard Fare: " +  S1.CALCULATE_FARE()
