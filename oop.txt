     ADD_CLAIM(claim) 
	SET Coverage TO claim.GET_COVERGAE()
	IF Coverage.GET_COVERAGE_LIMIT() GREATER THAN CLAIM.GET_AMOUNT() THEN 
	  DISPLAY "APPROVED" 
	  ADD claim TO LIST CLAIM
	ELSE 
	  DISPLAY "Limit Reached. DENIED" 
	END IF 
     END METHOD 

     ADD_COVERAGE(coverage)
	ADD coverage TO LIST COVERAGE 
     END METHOD 

     CALCULATE_REMAINING_COVERAGE() 
	SET APPROVED_CLAIMS TO 0 
	REPEAT FROM I=1 TO SIZE OF CLAIMS 
	SET claim TO CLAIMS[I]
	IF claim.GET_STATUS() EQUALS "APPROVED" THEN 
	 SET APPROVED_CLAIMS TO APPROVED_CLAIMS + claim.GET_AMOUNT() 
	END IF
	END REPEAT  
	SET TOTAL_COVERAGE TO 0 
	REPEAT FROM I=1 TO SIZE OF COVERAGE 
	 SET coverage TO COVERGAE[I] 
	 SET TOTAL_COVERAGE TO TOTAL_COVERAGE + coverage.GET_COVERAGE_LIMIT()
	END REPEAT 
	SET REMAINING_COVERAGE TO TOTAL_COVERAGE - APPROVED_CLAIMS
     END METHOD 
 END CLASS
