	  SET TOTAL_QUANTITY TO 0 
	  REPEAT FROM I=1 TO SIZE OF STRORAGE_BIN
	  SET bin TO STRORAGE_BIN[I] 
	  SET TOTAL_QUANTITY TO bin.GET_QUANTITY() + TOTAL_QUANTITY 
	  END REPEAT 
	  DISPLAY "TOTAL QUANTITY AVAILABLE FOR THIS PRODUCT IS:" + TOTAL_QUANTITY 
	END METHOD

	  SET STORAGE.GET_QUANTITY TO STORAGE.GET_QUANTITY + AMOUNT
	END METHOD 
	
	ADD_BIN(bin: tag) 
	  ADD bin TO LIST STRORAGE_BIN_NUMBER
	END METHOD 

	GET_LOCATION() 
	  DISPLAY STRORAGE_BIN_NUMBER
	END METHOD 
  END CLASS

	IS_FULL() 
	 IF SIZE OF STUDENTS LESS THAN GET_MAXIMUM_NUMBER THEN

	 ELSE 
	    RETURN TRUE
	 END IF

	ENROLL(Student, Time) 
	 REPEAT FROM I=1 TO SIZE OF STUDENTS  
	  IF STUDENTS[I] EQUALS Student THEN 
	   DISPLAY "Student already enrolled"
	  END IF 
	  END REPEAT 
	 
	   REPEAT FROM I=1 TO SIZE OF Student.ENROLLED_CLASSES
	     SET class to Student.ENROLLED_CLASSES[I]  
	   IF class.GET_CLASS_TIME() EQUALS Time THEN 
	     DIPLSY "Timing conflict, Can not enroll in this class" 	
            ELSE 
	      DISPLAY "ENROLLED"
	     STUDENTS.ADD_STUDENT(Student)  
	      ADD Student TO LIST STUDENTS
	      STUDENT.ADD_CLASS(Class) 
	        ADD Class TO ENROLLED_CLASSES
	   END IF 
	END REPEAT
	END METHOD

	SET ENROLLED_CLASSES TO LIST<TEXT>[] 
    END CONSTRUCTOR  
	
  METHOD
      ADD_GRADE(Mark: value) 
	ADD Mark TO LIST RECORDED_GRADES[] 
      END METHOD 

      CALCULATE_AVERAGE() 
	REPEAT FROM I=1 TO SIZE OF RECORDED_GRADES
	SET TOTAL_GRADES TO 0 
	SET TOTAL_WEIGHT TO 0 
	SET Mark to RECORDED_GRADES[I]
	SET TOTAL_GRADES TO TOTAL_GRADES + Mark.GET_CALCULATE_WEIGHTEDGRADE() 
	SET TOTAL_WEIGHT TO TOTAL_WEIGHT + Mark.GET_WEIGHTING()
	END REPEAT 
	SET AVERAGE TO TOTAL_GRADES / TOTAL_WEIGHT
	DISPLAY AVERAGE 
     END METHOD 
 END CLASS
