CLASS OWNER 
 ATTRIBUTES
	PRIVATE OWNER_ID: NUMBER  
 	PRIVATE NAME: TEXT 
	PRIVATE PhoneNumber: TEXT 
	PRIVATE PETS: LIST<TEXT>[] 
	
     CONSTRUSCTOR:(N:TEXT, NUM:TEXT, P:LIST) 
	SET NAME TO N 
	SET PhoneNumber TO NUM 
	SET PETS TO P 
     END CONSTRUCTOR 
 
 METHOD 
	GET_OWNER_ID() 
	  DISPLAY ID 
	END METHOD 
	SET_PETS(NEW_PET : PET) 
	 ADD NEW_PET TO LIST PETS
	END METHOD 
	SET_REMOVE_PETS(OLD_PET : PET)
	 REPEAT FROM I=1 TO SIZE OF LIST PETS 
	  IF PETS[I] EQUALS OLD_PET THEN  
	  REMOVE OLD_PET FROM LIST PETS
	  ELSE 
	  DISPLAY "PET NOT FOUND"
	  END IF 
	 END REPEAT  
	END METHOD 
 END CLASS 

 CLASS PET 
  ATTRIBUTES 
	PET_ID: NUMBER  
	PET_NAME: TEXT
	DOB: TEXT 
	SPECIES: TEXT 
	MEDICAL_HISTORY: LIST<TEXT>[] 
	
     CONSTRUCTOR:(P:TEXT, DB:TEXT, S:TEXT, MH:LIST<TEXT>) 
	SET PET_NAME TO P 
	SET DOB TO DB 
	SET SPECIES TO S 
	SET MEDICAL_HISTORY TO MH 
     END CONSTRUCTOR 

 METHOD 
	GET_PET_ID() 
	 DISPLAY PET_ID 
	END METHOD 
	GET_DOB() 
	 DISPLAY DOB 
	END METHOD 
	GET_SPECIES() 
	 DISPLAY SPECIES 
	END METHOD 
	SET_MEDICAL_HISTORY(MEDICAL_1 : NOTES) 
	 ADD MEDICAL_1 TO LIST MEDICAL_HISTORY
	END METHOD 
 END CLASS 

CLASS EXAMINATION
