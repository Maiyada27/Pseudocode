	END METHOD

     CONSTRUCTOR(TN:TEXT, FN:TEXT, ST:TEXT)
	SET TICKET_NUMBER TO TN 
	SET FILM_NAME TO FM  
	SET SHOW_TIME TO ST 
     END CONSTRUCTOR 

  METHOD 
	GET_TICKET_NUMBER() 
	  DISPLAY TICKET_NUMBER
 	END METHOD
	SET_SEAT_NUMBER(Seat: number)
	 READ Seat  
	 IF NumberAvailableSeats EQUALS 0 THEN 
	    DISPLAY "TICKETS SOLD OUT" 
	 ELSE 
	    REMOVE Seat FROM AvailableSeats
	    SET SEAT_NUMBER TO Seat
	    DISPLAY "Ticket Booked Successfully"  
	 END IF
	END METHOD 
END CLASS
