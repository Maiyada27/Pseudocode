CLASS CIRCLE INHERITS SHAPE 
   ATTRIBUTES
	PRIVATE RADIUS: NUMBER 
      CONSTRUCTOR(C:TEXT, PN:TEXT, A:NUMBER, P:NUMBER, R:NUMBER)
	SUPER(C, PN, A, P) 
	SET RADIUS TO R 
       END CONSTRUCTOR 

   METHOD 
	OVERRIDE CALCULATE_AREA()
	  SET AREA TO 3.14 * (RADIUS*RADIUS) 
	  RETURN AREA 
	END METHOD 

	OVERRIDE CALCULATE_PERIMETER() 
	 SET PERIMETER TO 2 * 3.14 * RADIUS 
	 RETURN PERIMETER 
	END METHOD 
  END CLASS 

CLASS RECTANGLE INHERITS SHAPE 
   ATTRIBUTES
	PRIVATE WIDTH: NUMBER
	PRIVATE HEIGHT: NUMBER 
      CONSTRUCTOR(C:TEXT, PN:TEXT, A:NUMBER, P:NUMBER, W:NUMBER, H:NUMBER)
	SUPER(C, PN, A, P) 
	SET WIDTH TO W
	SET HEIGHT TO H  
       END CONSTRUCTOR 

   METHOD 
	OVERRIDE CALCULATE_AREA()
	  SET AREA TO HEIGHT * WIDTH 
	  RETURN AREA 
	END METHOD 

	OVERRIDE CALCULATE_PERIMETER() 
	 SET PERIMETER TO (2 * HEIGHT) + (2 * WIDTH) 
	 RETURN PERIMETER 
	END METHOD 
  END CLASS 

CLASS TRIANGLE INHERITS SHAPE 
   ATTRIBUTES
	PRIVATE LENGTH_1: NUMBER
	PRIVATE LENGTH_2: NUMBER 
	PRIVATE LENGTH_3: NUMBER 
      CONSTRUCTOR(C:TEXT, PN:TEXT, A:NUMBER, P:NUMBER, L1:NUMBER, L2:NUMBER, L3:NUMBER)
	SUPER(C, PN, A, P) 
	SET BASE TO B
	SET LENGTH_1 TO L1
	SET LENGTH_2 TO L2
      END CONSTRUCTOR 

   METHOD 
	OVERRIDE CALCULATE_AREA()
	  SET S TO (LENGTH_1 + LENGTH_2 + LENGTH_3) / 2
	  SET AREA TO SQRT(S*(S-LENGTH_1) * S*(S-LENGTH_2) * S*(S-LENGTH_3))  
	  RETURN AREA 
	END METHOD 

	OVERRIDE CALCULATE_PERIMETER() 
	 SET PERIMETER TO LENGTH_1 + LENGTH_2 + LENGTH_3 
	 RETURN PERIMETER 
	END METHOD 
  END CLASS

CLASS ORDER 
	PRIVATE ORDER_ID: NUMBER 
	PRIVATE ITEMS:LIST<ITEM>[] 
	PRIVATE PAYMENT: PAYMENT 
	
    CONSTRUCTOR()  
	SET ITEMS TO LIST[] 
    END CONSTRUCTOR 

  METHOD 
 	ADD_ITEM(item) 
	   ADD item TO LIST ITEMS 
	END METHOD 
	
	REMOVE_ITEM(item) 
	   REMOVE item FROM LIST ITEMS 
	END METHOD 

	CALCULATE_TOTAL() 
	 SET TOTAL TO 0
	   REPEAT FROM I=0 TO SIZE OF ITEMS - 1 
	   SET TOTAL TO TOTAL + ITEMS[I].GET_COST() 
	   END REPEAT 
           RETURN TOTAL 
	END METHOD 

	SET_PAYMENT(payment: text) 
	  SET PAYMENT TO payment 
	END METHOD

	CHECKOUT() 
	  PAYMENT.PROCESS_PAYMENT()
	END METHOD  
	
  END CLASS 

START 
   BEGIN
	CREATE OBJET item1 OF ITEM("Iced Latte", 2) 
	CREATE OBJET item2 OF ITEM("Flat white", 2)  
	CREATE OBJECT order1 OF ORDER() 

	order1.ADD_ITEM(item1)
	order1.ADD_ITEM(item2)
	order1.CALCULATE_TOTAL()
	
	CREATE OBJET payment1 OF CREDIT_CARD_PAYMENT(order1.CALCULATE_TOTAL(), "4431-7765-2234-8888", "05/28") 
	
	order1.SET_PAYMENT(payment1) 
	order1.CHECKOUT()
   END

    PRIVATE UNPAID_PURCHASES: LIST<NUMBER>[]  

   CONSTRUCTOR(CN:TEXT, ED:TEXT, S:TEXT, C:CHIP, CL:NUMBER) 
	SUPER(CN, ED, S, C) 
	SET CREDIT_LIMIT TO CL
	SET UNPAID_PURCHASES TO LIST[]
   END CONSTRUCTOR 

  METHOD
	CALCULATE_OUTSTANDING_BALANCE()
	SET TOTAL TO 0 
	 REPEAT FROM I=0 TO SIZE OF UNPAID_PURCHASES - 1
	  SET TOTAL TO TOTAL + UNPAID_PURCHASES[I] 
	 END REPEAT 
	END METHOD
	
	OVERRIDE AUTHORIZE_PURCHASE(Amount: NUMBER)
	 IF Amount + CALCULATE_OUTSTANDING_BALANCE() LESS THAN OR EQUAL TO CREDIT_LIMIT THEN 
	 DISPLAY "Successful" 
	 ADD Amount to UNPAID_PURCHASES
	 ELSE 
	  DISPLAY "Exceeded credit limit" 
	 END IF 
	END METHOD
 END CLASS
