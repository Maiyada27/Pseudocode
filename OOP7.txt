CLASS CLAIM
  ATTRIBUTES
	PRIVATE CLAIM_ID: NUMBER
    	PRIVATE CLAIM_NUMBER: TEXT
    	PRIVATE DATE_FILED: TEXT
    	PRIVATE STATUS: TEXT
    	PRIVATE FOLDER: DOCUMENTATION_FOLDER

    CONSTRUCTOR(CN:TEXT, DF:TEXT, F:DOCUMENTATION_FOLDER)
    	SET CLAIM_NUMBER TO CN
    	SET DATE_FILED TO DF
    	SET STATUS TO "Filed"
    	SET FOLDER TO F
    END CONSTRUCTOR


 METHOD
    	UPDATE_STATUS(NEW_STATUS:TEXT)
         IF STATUS EQUALS "Filed" AND NEW_STATUS EQUALS "Under Review" THEN
            SET STATUS TO NEW_STATUS
         ELSE IF STATUS EQUALS "Under Review" AND NEW_STATUS EQUALS "Approved" THEN
            SET STATUS TO NEW_STATUS
         ELSE IF STATUS EQUALS "Approved" AND NEW_STATUS EQUALS "Closed" THEN
            SET STATUS TO NEW_STATUS
         ELSE
            DISPLAY "Invalid status transition."
         END IF
    	END METHOD

    	CALCULATE_PAYOUT()
          DISPLAY "Payout depends on claim type."
    	END METHOD
END CLASS

CLASS DATABASE_SERVER INHERITS SERVER
 ATTRIBUTES
    	PRIVATE CURRENT_QUERIES: NUMBER
    	PRIVATE MAX_QUERIES: NUMBER

   CONSTRUCTOR(SN:TEXT, U:NUMBER, H:HARDWARE_SPECIFICATION, CQ:NUMBER, MQ:NUMBER)
    	SUPER(SN, U, H)
    	SET CURRENT_QUERIES TO CQ
    	SET MAX_QUERIES TO MQ
   END CONSTRUCTOR

 METHOD
    	OVERRIDE CALCULATE_LOAD()
          SET LOAD TO (CURRENT_QUERIES / MAX_QUERIES) * 100
          RETURN LOAD
    	END METHOD

    	ACCEPT_CONNECTION()
          IF CURRENT_QUERIES GREATER THAN OR EQUAL TO MAX_QUERIES THEN
            DISPLAY "Server reached full capacity. Connection failed."
          ELSE
            DISPLAY "Connection accepted."
          END IF
	END METHOD
END CLASS


CLASS CACHE_SERVER INHERITS DATABASE_SERVER
 ATTRIBUTES
    	PRIVATE CACHE_HIT_RATIO: NUMBER

   CONSTRUCTOR(SN:TEXT, U:NUMBER, H:HARDWARE_SPECIFICATION, CQ:NUMBER, MQ:NUMBER, CHR:NUMBER)
    	SUPER(SN, U, H, CQ, MQ)
    	SET CACHE_HIT_RATIO TO CHR
   END CONSTRUCTOR

 METHOD
    	OVERRIDE CALCULATE_LOAD()
          SET LOAD TO (CURRENT_QUERIES / MAX_QUERIES) * 100
          SET EFFECTIVE_LOAD TO LOAD - (LOAD * CACHE_HIT_RATIO / 100)
          RETURN EFFECTIVE_LOAD
    	END METHOD
END CLASS
