CLASS BOOK
  ATTRIBUTES 
	PRIVATE BOOK_ID: NUMBER 
	PRIVATE BOOK_TITLE: TEXT
	PRIVATE BOOK_AUTHOR: TEXT  
	PRIVATE BOOK_GENRE: TEXT
	PRIVATE BOOK_COST: NUMBER
	  
   CONSTRUCTOR(BT:TEXT, BA:TEXT,BG:TEXT, BC:NUMBER) 
	SET BOOK_TITLE TO BT 
	SET BOOK_AUTHOR TO BA 
	SET BOOK_GENRE TO BG 
	SET BOOK_COST TO BC 
   END CONSTRUCTOR 

  METHOD 
	GET_BOOK_TITLE() 
	   DISPLAY BOOK_TITLE 
	END METHOD 
	
	GET_COST() 
	   DISPLAY BOOK_COST 
	END METHOD 
  END CLASS 
 	 
CLASS CUSTOMER
  ATTRIBUTES 
	PRIVATE CUSTOMER_ID: NUMBER 
	PRIVATE CUSTOMER_NAME: TEXT
	PRIVATE PURCHASE_HISTORY:LIST<TEXT>[] 

   CONSTRUCTOR(CN:TEXT) 
	SET CUSTOMER_NAME TO CN 
	SET PURCHASE_HISTORY TO LIST[] 
   END CONSTRUCTOR 

  METHOD 
	ADD_TO_PURCHASE_HISTORY(book)
	  ADD book TO LIST PURCHASE_HISTORY
	END METHOD 

	GET_PURCHASE_HISTORY() 
	   DISPLAY PURCHASE_HISTORY
	END METHOD 
  END CLASS 

CLASS SALE
  ATTRIBUTES  
	PRIVATE SALE_ID: NUMBER  
	PRIVATE CUSTOMER: TEXT 
	PRIVATE BASKET: LIST<TEXT>[] 
	PRIVATE TOTAL_PRICE: NUMBER 

   CONSTRUCTOR(C:TEXT) 
	SET CUSTOMER TO C 
	SET BASKET TO LIST[] 
   END CONSTRUCTOR 

  METHOD 
	ADD_BOOK_TO_BASKET(book: new) 
	  ADD book TO LIST BASKET 
	END METHOD 
	REMOVE_BOOK_FROM_BASKET(book: new) 
	  REMOVE book FROM LIST BASKET 
	END METHOD 
	CALCULATE_TOTAL_PRICE() 
	   SET TOTAL TO 0 
	   REPEAT FROM I=1 TO SIZE OF BASKET 
	   SET book TO BASKET[I] 
	   SET TOTAL TO TOTAL + book.GET_COST() 
	   END REPEAT 
	END METHOD 
  END CLASS 

START 
   BEGIN 
	CREATE OBJECT book1 OF BOOK("Crime and punishment", "Dostoevsky", "Literature", 7) 
	CREATE OBJECT book2 OF BOOK("The years", "Annie Ernoux", "Literature", 5) 
	CREATE OBJECT customer1 OF CUSTOMER("Mays Mohamed") 
	CREATE OBJECT sale1 OF SALE(customer1) 

	sale1.ADD_BOOK_TO_BASKET(book1) 
	sale1.ADD_BOOK_TO_BASKET(book2)
	customer1.ADD_TO_PURCHASE_HISTORY(book1) 
	customer1.ADD_TO_PURCHASE_HISTORY(book2) 

	DISPLAY "TOTAL COST: " + sale1.CALCULATE_TOTAL_PRICE()
     END 
  STOP
