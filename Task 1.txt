CLASS HOSPITAL 
  ATTRIBUTES 
	ID: NUMBER 
 	HOSPITAL_NAME: TEXT 
	ADDRESS: TEXT 
	PATIENTS: LIST <TEXT>[] 

   CONSTRUCTOR(PL: LIST<TEXT>) 
	SET PATIENTS TO P 
   END CONSTRUCTOR 
	
   METHODS
	ADMIT PATIENT (NAME : PATIENT)
	ADD NAME TO LIST PATIENTS 
  	END METHOD 
	DISCHARGE PATIENT (NAME: PATIENT) 
	REMOVE NAME FROM LIST PATIENTS   
	END METHOD
	COUNT ADMITTED PATIENTS  
	DISPLAY SIZE OF LIST PATIENTS 
	END METHOD 
 END CLASS

	NAME: TEXT

	DISCHARGE_DATE: TET 
	ROOM: NUMBER

	SET DOB TO B

	SET DISCHARGE_DATE TO DD 
	SET ROOM TO R 
	SET PROCEDURE TO P

	END METHOD

	END METHOD

 END CLASS

 END CLASS
