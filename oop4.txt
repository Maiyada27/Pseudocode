	SET BASKET TO LIST[] 
   END CONSTRUCTOR 

  METHOD 
	ADD_BOOK_TO_BASKET(book: new) 
	  ADD book TO LIST BASKET 
	END METHOD 
	REMOVE_BOOK_FROM_BASKET(book: new) 
	  REMOVE book FROM LIST BASKET 
	END METHOD 
	CALCULATE_TOTAL_PRICE() 
	   SET TOTAL TO 0 
	   REPEAT FROM I=1 TO SIZE OF BASKET 
	   SET book TO BASKET[I]

	   END REPEAT
	   RETURN TOTAL  
	END METHOD 
  END CLASS

     CONSTRUCTOR(S:TEXT, E:TEXT) 	
	SET STUDENT TO S 
	SET EXAM TO E 
	SET SCORE TO 0 
	SET ANSWERS TO LIST[]
	SET ELAPSED_TIME TO 0
     END CONSTRUCTOR 
  
    METHOD 
	ADD_ANSWER(answer)
	  IF ELAPSED_TIME GREATER THAN OR EQUAL TO EXAM.GET_DURATION() THEN 
	    DISPLAY "Time is over" 
	  ELSE  
          ADD answer TO LIST ANSWERS 
	END METHOD 

	CALCULATE_SCORE() 
	  SET SCORE TO 0 
	  REPEAT FROM I=0 TO SIZE OF ANSWERS - 1
	   SET answer TO ANSWER[I] 
	   IF answer.GET_QUESTION().GET_SOLUTION() equals answer.GET_SELECTED_SOLUTION() THEN 
	     SET SCORE TO SCORE + question.GET_MARK()  
	   END IF 
	  END REPEAT 
	  RETURN SCORE 
	 END METHOD 
    END CLASS

	ADD_BOOKING(booking) 
	  REPEAT FROM I=0 TO SIZE OF BOOKINGS - 1
	  IF booking[I].GET_TIME_SLOT() EQUALS booking.GET_TIME_SLOT() THEN

	  END IF 
	  END REPEAT

	  RETURN TRUE
	END METHOD

CLASS BOOKING 
  ATTRIBUTES
	PRIVATE BOOKING_ID: NUMBER
	PRIVATE MEMBER: MEMBER  
	PRIVATE MEETING_ROOM: ROOM 
	PRIVATE TIME_SLOT: TEXT 
	PRIVATE DURATION: NUMBER 
  
   CONSTRUCTOR(M:TEXT, MR:NUMBER, T:TEXT, D:NUMBER)
	SET THIS.MEMBER TO M  
	SET THIS.MEETING_ROOM TO MR 
	SET THIS.TIME_SLOT TO T 
	SET THIS.DURATION TO D 
   END CONSTRUCTOR 

  METHOD 
	GET_DURATION() 
	   RETURN DURATION 
	END METHOD 
	GET_TIME_SLOT() 
	    RETURN TIME 
	END METHOD 

	MAKE_BOOKING()  
	IF MEMBER.CALCULATE_REMAINING_HOURS() LESS THAN DURATION THEN 
	  DISPLAY "Failed to book. Remaining hours exceeded"
	ELSE 
	  IF MEETING_ROOM.ADD_BOOKING(SELF) EQUALS TRUE THEN 
	  MEMBER.ADD_BOOKING()
	  DISPLAY "Successfully Booked"  
	  END IF 
	END IF 
	END METHOD 
  END CLASS

CLASS PERSONAL_TRAINER 
  ATTRIBUTES 
	PRIVATE PERSONAL_TRAINER_ID: NUMBER 
	PRIVATE PT_NAME: TEXT 
	PRIVATE SESSIONS: LIST<TEXT>[] 
	
    CONSTRUCTOR(PN:TEXT) 
	SET THIS.PT_NAME TO PN 
	SET THIS.SESSIONS TO LIST[] 
    END CONSTRUCTOR 
	
  METHOD
	ADD_SESSION(session) 
	 REPEAT FROM I=0 TO SIZE OF SESSIONS - 1
	 IF SESSIONS[I].GET_TIME() EQUALS session.GET_TIME() AND SESSIONS[I].GET_DATE() EQUALS session.GET_DATE()  	 THEN 
	  "Failed. Time already booked for another client" 
	   RETURN FALSE 
	 ELSE 	
	   ADD session TO LIST SESSIONS 
	  RETURN TRUE 
	 END IF 
	 END REPEAT  
	END METHOD

	GET_SESSIONS() 
	   RETURN SESSIONS
	END METHOD 
	
  END CLASS
