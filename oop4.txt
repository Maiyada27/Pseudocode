  CLASS ATTEMPT
   ATTRIBUTES 
	PRIVATE ATTEMPT_ID: NUMBER 
	PRIVATE STUDENT: TEXT
	PRIVATE EXAM: TEXT  
	PRIVATE ANSWERS: LIST<TEXT>[]
	PRIVATE SCORE: number

     CONSTRUCTOR(S:TEXT, E:TEXT) 	
	SET STUDENT TO S 
	SET EXAM TO E 
	SET SCORE TO 0 
	SET ANSWERS TO LIST[]
     END CONSTRUCTOR 
  
    METHOD 
	ADD_ANSWER(answer) 
          ADD answer TO LIST ANSWERS 
	END METHOD 
	CALCULATE_SCORE() 
	  SET SCORE TO 0 
	  REPEAT FROM I=0 TO SIZE OF ANSWERS - 1
	   SET answer TO ANSWER[I]
	   SET question TO EXAM.GET_QUESTIONS()[I] 
	   IF answer equals question.GET_SOLUTION() THEN 
	     SET SCORE TO SCORE + question.GET_MARK()  
	   END IF 
	  END REPEAT 
	  RETURN SCORE 
	 END METHOD 
    END CLASS 

START 
   BEGIN 
	CREATE OBJECT exam1 OF EXAM("STATISTICS1", "2 HOURS") 
	CREATE OBJECT question1 OF QUESTION("Calculate the mean", "122", 25)
	CREATE OBJECT question2 OF QUESTION("Calculate the standard deviation", "84", 50)
	CREATE OBJECT student1 OF STUDENT("Mahmood Ali", "Mahmood@uni.com") 
	CREATE OBJECT attempt1 OF ATTEMPT(student1, exam)

	exam1.ADD_QUESTION(question1) 
	exam1.ADD_QUESTION(question2)
	student1.ADD_ATTEMPT(attempt1)
	attempt1.ADD_ANSWER("122") 
	attempt1.ADD_ANSWER("80")  

	DISPLAY "Exam Score = " + attempt1.CALCULATE_SCORE()
     END 
  STOP

CLASS BOOKING 
  ATTRIBUTES
	PRIVATE BOOKING_ID: NUMBER
	PRIVATE MEMBER: TEXT  
	PRIVATE MEETING_ROOM: NUMBER 
	PRIVATE TIME_SLOT: TEXT 
	PRIVATE DURATION: NUMBER 
  
   CONSTRUCTOR(M:TEXT, MR:NUMBER, T:TEXT, D:NUMBER)
	SET MEMBER TO M  
	SET MEETING_ROOM TO MR 
	SET TIME_SLOT TO T 
	SET DURATION TO D 
   END CONSTRUCTOR 

  METHOD 
	GET_DURATION() 
	   RETURN DURATION 
	END METHOD 
	GET_TIME_SLOT() 
	    RETURN TIME 
	END METHOD 

	MAKE_BOOKING()  
	IF MEMBER.CALCULATE_REMAINING_HOURS() LESS THAN DURATION THEN 
	  DISPLAY "Failed to book. Remaining hours exceeded"
	ELSE 
	  IF MEETING_ROOM.ADD_BOOKING() EQUALS TRUE THEN 
	  MEMBER.ADD_BOOKING()
	  DISPLAY "Successfully Booked"  
	  END IF 
	END IF 
	END METHOD 
  END CLASS

CLASS PERSONAL_TRAINER 
  ATTRIBUTES 
	PRIVATE PERSONAL_TRAINER_ID: NUMBER 
	PRIVATE PT_NAME: TEXT 
	PRIVATE SESSIONS: LIST<TEXT>[] 
	
    CONSTRUCTOR(PN:TEXT) 
	SET PT_NAME TO PN 
	SET SESSIONS TO LIST[] 
    END CONSTRUCTOR 
	
  METHOD
	ADD_SESSION(session) 
	 REPEAT FROM I=0 TO SIZE OF SESSIONS - 1
	 IF SESSION[I].GET_TIME() EQUALS session.GET_TIME() AND SESSION[I].GET_DATE() EQUALS session.GET_DATE()  	 THEN 
	  "Failed. Time already booked for another client" 
	   RETURN FALSE 
	 ELSE 	
	   ADD session TO LIST SESSIONS 
	  RETURN TRUE 
	 END IF 
	 END REPEAT  
	END METHOD

	GET_SESSIONS() 
	   RETURN SESSIONS
	END METHOD 
	
  END CLASS
